string Except(string s,int x)  
{  
    int cmp=0,ok=0;  
    string ans="";  
    for(int i=0;i<s.size();i++)  
    {  
        cmp=(cmp*10+s[i]-'0');  
        if(cmp>=x)  
        {  
            ok=1;  
            ans+=(cmp/x+'0');  
            cmp%=x;  
        }  
        else{  
            if(ok==1)  
                ans+='0';    
        }  
    }  
    return ans;  
}
